Business Applications

Professional scheduling and appointment booking

Business Hours Only

Restrict selections to weekdays during business hours

InfiDate.create('#business-hours', {
  mode: 'single',
  disable: [
    function(date) {
      // Disable weekends
      return date.getDay() === 0 || date.getDay() === 6;
    },
    // Disable holidays
    '2025-12-25', '2025-01-01', '2025-07-04'
  ],
  minDate: 'today',
  displayFormat: 'dddd, MMMM D, YYYY'
});

Meeting Scheduler

Advanced scheduling with time slots and availability

InfiDate.create('#meeting-scheduler', {
  mode: 'single',
  displayMode: 'modal',
  title: 'Schedule Meeting',
  subtitle: 'Select available date',
  disable: [
    function(date) {
      // Disable weekends and past dates
      const today = new Date();
      return date < today || date.getDay() === 0 || date.getDay() === 6;
    }
  ],
  onChange: function(data) {
    showTimeSlots(data.date);
  }
});

Project Deadlines

Set project milestones with minimum lead time

const minDeadline = new Date();
minDeadline.setDate(minDeadline.getDate() + 7); // 1 week minimum

InfiDate.create('#project-deadline', {
  mode: 'single',
  minDate: minDeadline,
  displayFormat: 'MMMM D, YYYY',
  onChange: function(data) {
    const daysUntil = Math.ceil((data.date - new Date()) / (1000 * 60 * 60 * 24));
    console.log(`Deadline set: ${daysUntil} days from now`);
  }
});

Travel & Booking

Hotel bookings, flight dates, and vacation planning

Hotel Booking

Check-in and check-out with minimum stay requirements

InfiDate.create('#hotel-booking', {
  mode: 'range',
  displayMode: 'modal',
  title: 'Hotel Booking',
  subtitle: 'Select check-in and check-out dates',
  minDate: 'today',
  minRangeDays: 2, // Minimum 2-night stay
  maxRangeDays: 30, // Maximum 30-night stay
  onChange: function(data) {
    if (data.start && data.end) {
      const nights = data.nights;
      const total = nights * 150; // $150 per night
      document.getElementById('hotel-result').innerHTML = 
        `Booking: ${nights} nights - Total: $${total}`;
    }
  }
});

Flight Booking

Departure and return dates with pricing

InfiDate.create('#flight-booking', {
  mode: 'range',
  minDate: 'today',
  maxRangeDays: 365,
  displayFormat: 'MMM D, YYYY',
  onChange: function(data) {
    if (data.start && data.end) {
      const isWeekend = data.start.getDay() === 0 || data.start.getDay() === 6;
      const basePrice = isWeekend ? 450 : 350;
      document.getElementById('flight-result').innerHTML = 
        `Flight: ${data.formatted.start} - ${data.formatted.end}

         Price: $${basePrice} (${isWeekend ? 'Weekend' : 'Weekday'} rate)`;
    }
  }
});
